function amplitude=pamplitude(n,m,t00,tnm,a00,tm)
    amplitude2=((a00.^2)./(n.^2+m.^2+2.*(n+m)+2)) ;
    t1=tm-(t00/(4*pi)).*sin(4*pi*tm/t00);
    t2=tm-(tnm/(4*pi)).*sin(4*pi*tm/tnm);
    amplitude=sqrt(2*amplitude2*t1/t2);
